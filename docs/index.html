Basic Example

import vwebx

struct App {
    vwebx.App
}

fn new_app() &App {
    mut app := &App{
        App: vwebx.new_app()
    }
    return app
}

fn (mut app App) index() vwebx.Result {
    return app.json({
        'message': 'Hello, VWebX!'
    })
}

fn main() {
    mut app := new_app()
    app.get('/', app.index)
    app.run()
}
